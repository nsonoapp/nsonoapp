rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {

    // ── Helpers rapides (sans get) ─────────────────────────────
    function signedIn() {
      return request.auth != null;
    }

    function validUserRole(role) {
      return role in ['admin', 'seller', 'user'];
    }

    function validBusinessRole(role) {
      return role in ['admin', 'seller'];
    }

    // ── get() UNIQUEMENT ici (users / meta / admin) ────────────
    function metaDoc() {
      return get(/databases/$(database)/documents/system/meta);
    }

    function metaExists() {
      return exists(/databases/$(database)/documents/system/meta);
    }

    function validMetaLimits(data) {
      return data.maxUsers is int
        && data.maxUsers >= 1
        && data.maxUsers <= 500
        && data.usersCount is int
        && data.usersCount >= 0
        && data.usersCount <= data.maxUsers;
    }

    function underUserLimit() {
      return metaExists()
        && metaDoc().data.usersCount < metaDoc().data.maxUsers;
    }

    function canIncrementUsersCount() {
      return signedIn()
        && metaExists()
        && request.resource.data.usersCount == resource.data.usersCount + 1
        && request.resource.data.maxUsers == resource.data.maxUsers
        && request.resource.data.usersCount <= resource.data.maxUsers;
    }

    function canCreateOwnLog() {
      return signedIn()
        && request.resource.data.userId == request.auth.uid
        && request.resource.data.keys().hasAll(['userId', 'action', 'createdAt']);
    }

    function isAdmin() {
      return signedIn()
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }

    function isActiveMember() {
      return signedIn()
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.isActive == true
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.approvalStatus == 'approved'
        && validBusinessRole(get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role);
    }

    function currentUserDoc() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid));
    }

    function hasScopedCompany(data) {
      return !data.keys().hasAny(['companyId'])
        || data.companyId == null
        || (
          currentUserDoc().data.companyId is string
          && data.companyId == currentUserDoc().data.companyId
        )
        || (
          data.companyId == 'main'
          && isMasterOfCompany('main')
        );
    }

    function hasScopedEntity(data) {
      return !data.keys().hasAny(['entityId'])
        || data.entityId == null
        || isMasterOfCompany('main')
        || (
          currentUserDoc().data.entityId is string
          && data.entityId == currentUserDoc().data.entityId
        );
    }

    function canReadBusinessDoc(data) {
      return isActiveMember()
        && hasScopedCompany(data)
        && hasScopedEntity(data);
    }

    function canWriteBusinessDoc(data) {
      return isActiveMember()
        && hasScopedCompany(data)
        && hasScopedEntity(data);
    }

    function companyExists() {
      return exists(/databases/$(database)/documents/companies/main);
    }

    function validCompanyLink() {
      return !request.resource.data.keys().hasAny(['companyId'])
        || request.resource.data.companyId == null
        || (
          request.resource.data.companyId == 'main'
          && companyExists()
        );
    }

    function userBelongsToCompany() {
      return !resource.data.keys().hasAny(['companyId'])
        || resource.data.companyId == null
        || !request.resource.data.keys().hasAny(['companyId'])
        || request.resource.data.companyId == resource.data.companyId;
    }

    function isMasterOfCompany(companyId) {
      return companyId == 'main'
        && companyExists()
        && (
          get(/databases/$(database)/documents/companies/main).data.masterAdminId == request.auth.uid
          || (
            get(/databases/$(database)/documents/companies/main).data.masterAdminIds is list
            && request.auth.uid in get(/databases/$(database)/documents/companies/main).data.masterAdminIds
          )
        );
    }

    function validUserCreate() {
      return request.resource.data.userId == request.auth.uid
        && request.resource.data.keys().hasAll(['userId', 'name', 'role', 'isActive', 'createdAt'])
        && validUserRole(request.resource.data.role)
        && request.resource.data.isActive is bool
        && validCompanyLink();
    }

    // Inscription : role=user, pending, inactif jusqu'à approbation admin
    function validSignupUserCreate() {
      return request.resource.data.userId == request.auth.uid
        && request.resource.data.keys().hasAll(['userId', 'name', 'role', 'isActive', 'createdAt', 'approvalStatus'])
        && request.resource.data.role == 'user'
        && request.resource.data.isActive == false
        && request.resource.data.approvalStatus == 'pending'
        && validCompanyLink();
    }

    function validMasterAdminList(data) {
      return data.keys().hasAny(['masterAdminIds'])
        && data.masterAdminIds is list
        && data.masterAdminIds.size() > 0
        && data.masterAdminIds.size() <= 5;
    }
    // system/meta → { usersCount: 0, maxUsers: 100 } (quota société, pas boutique)
    match /system/meta {
      allow read: if signedIn();

      allow create: if signedIn()
        && validMetaLimits(request.resource.data)
        && request.resource.data.usersCount == 0;

      allow update: if signedIn()
        && validMetaLimits(request.resource.data)
        && (
          canIncrementUsersCount()
          || request.resource.data.usersCount == resource.data.usersCount - 1
          || isAdmin()
          || isMasterOfCompany('main')
        );
    }

    // Monitoring budget Firebase — lecture master admin uniquement
    match /system/budget_monitor {
      allow read: if signedIn() && isMasterOfCompany('main');
      allow write: if isMasterOfCompany('main');
    }

    // ── Users ──────────────────────────────────────────────────
    match /users/{userId} {
      allow read: if signedIn() && (request.auth.uid == userId || isActiveMember());

      // Création après Auth Firebase (signup) — quota via system/meta
      allow create: if signedIn()
        && request.auth.uid == userId
        && underUserLimit()
        && (validUserCreate() || validSignupUserCreate());

      allow update: if signedIn() && (
        (request.auth.uid == userId
          && request.resource.data.role == resource.data.role
          && userBelongsToCompany())
        || isAdmin()
      );

      allow delete: if isAdmin();
    }

    // ── Collections métier — auth seulement = rapide, 0 get ────
    match /products/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /sales/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /sale_items/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /stock_movements/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /purchases/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /purchase_items/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /expenses/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /debts/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /losses/{id} {
      allow read: if canReadBusinessDoc(resource.data);
      allow create, update: if canWriteBusinessDoc(request.resource.data);
      allow delete: if canWriteBusinessDoc(resource.data);
    }

    match /logs/{id} {
      allow read: if canReadBusinessDoc(resource.data)
        && validBusinessRole(currentUserDoc().data.role);
      allow create: if canCreateOwnLog()
        || (isActiveMember() && canWriteBusinessDoc(request.resource.data));
      allow update: if isActiveMember() && canWriteBusinessDoc(request.resource.data);
      allow delete: if isActiveMember() && canWriteBusinessDoc(resource.data);
    }

    // ── Config — lecture rapide, écriture admin ────────────────
    match /appConfig/{id} {
      allow read: if isActiveMember();
      allow write: if isAdmin();
    }

    match /settings/{id} {
      allow read: if isActiveMember();
      allow write: if isAdmin();
    }

    // ── NSONO — une seule société par base (companies/main) ────
    match /companies/{id} {
      // Requis pour le flux signup/login après Auth Firebase
      allow read: if signedIn();
      allow create: if signedIn()
        && id == 'main'
        && !companyExists()
        && validMasterAdminList(request.resource.data);
      allow update: if id == 'main' && (
        (
          isAdmin()
          || resource.data.masterAdminId == request.auth.uid
        )
        && validMasterAdminList(request.resource.data)
      );
      allow delete: if false;
    }

    match /entities/{id} {
      // Requis pour résolution entité durant signup/login
      allow read: if signedIn();
      allow create: if signedIn()
        && companyExists()
        && request.resource.data.companyId == 'main';
      allow update, delete: if isAdmin()
        && resource.data.companyId == 'main';
    }

    match /roles/{id} {
      allow read: if isActiveMember();
      allow create: if isAdmin()
        && request.resource.data.companyId == 'main';
      allow update: if isAdmin()
        && resource.data.companyId == 'main'
        && request.resource.data.companyId == 'main';
      allow delete: if isAdmin()
        && resource.data.companyId == 'main';
    }

    // ── Secret société — jamais lisible côté client ────────────
    match /company_secrets/{companyId} {
      allow read: if false;
      allow create: if signedIn()
        && companyId == 'main'
        && request.resource.data.companyId == 'main'
        && request.resource.data.keys().hasAll(['companyId', 'passwordHash', 'updatedAt'])
        && !exists(/databases/$(database)/documents/company_secrets/main);
      allow update: if companyId == 'main' && (isAdmin() || isMasterOfCompany('main'));
      allow delete: if false;
    }

    match /entity_secrets/{entityId} {
      allow read: if false;
      allow create: if signedIn()
        && request.resource.data.entityId == entityId
        && request.resource.data.companyId == 'main'
        && request.resource.data.keys().hasAll(['companyId', 'entityId', 'passwordHash', 'updatedAt'])
        && !exists(/databases/$(database)/documents/entity_secrets/$(entityId));
      allow update: if isAdmin() || isMasterOfCompany('main');
      allow delete: if false;
    }

    // ── Sonde auth — compare hash sans exposer company_secrets ─
    match /company_auth_probes/{probeId} {
      allow read: if false;
      allow create: if signedIn()
        && request.resource.data.companyId == 'main'
        && request.resource.data.passwordHash is string
        && request.resource.data.uid == request.auth.uid
        && exists(/databases/$(database)/documents/company_secrets/main)
        && get(/databases/$(database)/documents/company_secrets/main).data.passwordHash
           == request.resource.data.passwordHash;
      allow delete: if signedIn() && resource.data.uid == request.auth.uid;
      allow update: if false;
    }

    match /entity_auth_probes/{probeId} {
      allow read: if false;
      allow create: if signedIn()
        && request.resource.data.entityId is string
        && request.resource.data.passwordHash is string
        && request.resource.data.uid == request.auth.uid
        && exists(/databases/$(database)/documents/entity_secrets/$(request.resource.data.entityId))
        && get(/databases/$(database)/documents/entity_secrets/$(request.resource.data.entityId)).data.passwordHash
           == request.resource.data.passwordHash;
      allow delete: if signedIn() && resource.data.uid == request.auth.uid;
      allow update: if false;
    }
  }
}
